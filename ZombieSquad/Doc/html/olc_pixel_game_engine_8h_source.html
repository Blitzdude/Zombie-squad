9 What is this?
10 ~~~~~~~~~~~~~
11 The olcConsoleGameEngine has been a surprising and wonderful success for me,
12 and I'm delighted how people have reacted so positively towards it, so thanks
13 for that.
14 
15 However, there are limitations that I simply cannot avoid. Firstly, I need to
16 maintain several different versions of it to accommodate users on Windows7,
17 8, 10, Linux, Mac, Visual Studio & Code::Blocks. Secondly, this year I've been
18 pushing the console to the limits of its graphical capabilities and the effect
19 is becoming underwhelming. The engine itself is not slow at all, but the process
20 that Windows uses to draw the command prompt to the screen is, and worse still,
21 it's dynamic based upon the variation of character colours and glyphs. Sadly
22 I have no control over this, and recent videos that are extremely graphical
23 (for a command prompt :P ) have been dipping to unacceptable framerates. As
24 the channel has been popular with aspiring game developers, I'm concerned that
25 the visual appeal of the command prompt is perhaps limited to us oldies, and I
26 dont want to alienate younger learners. Finally, I'd like to demonstrate many
27 more algorithms and image processing that exist in the graphical domain, for
28 which the console is insufficient.
29 
30 For this reason, I have created olcPixelGameEngine! The look and feel to the
31 programmer is almost identical, so all of my existing code from the videos is
32 easily portable, and the programmer uses this file in exactly the same way. But
33 I've decided that rather than just build a command prompt emulator, that I
34 would at least harness some modern(ish) portable technologies.
35 
36 As a result, the olcPixelGameEngine supports 32-bit colour, is written in a
37 cross-platform style, uses modern(ish) C++ conventions and most importantly,
38 renders much much faster. I will use this version when my applications are
39 predominantly graphics based, but use the console version when they are
40 predominantly text based - Don't worry, loads more command prompt silliness to
41 come yet, but evolution is important!!

728  {

730  // Use GDI+

732 #ifdef __MINGW32__
733  wchar_t *buffer = new wchar_t[sImageFile.length() + 1];
734  mbstowcs(buffer, sImageFile.c_str(), sImageFile.length());
735  buffer[sImageFile.length()] = L'\0';
736  wsImageFile = buffer;
737  delete[] buffer;
738 #else
739  wsImageFile = ConvertS2W(sImageFile);
740 #endif
741  Gdiplus::Bitmap *bmp = Gdiplus::Bitmap::FromFile(wsImageFile.c_str());
742  if (bmp == nullptr)
743  return olc::NO_FILE;
744 
745  width = bmp->GetWidth();
746  height = bmp->GetHeight();

750  for (int y = 0; y < height; y++)

753  bmp->GetPixel(x, y, &c);
754  SetPixel(x, y, Pixel(c.GetRed(), c.GetGreen(), c.GetBlue(), c.GetAlpha()));
755  }
756  delete bmp;
757  return olc::OK;
758 #else
759  // Use libpng, Thanks to Guillaume Cottenceau
761  // https://gist.github.com/niw/5963798
762  png_structp png;
763  png_infop info;
764 
765  FILE *f = fopen(sImageFile.c_str(), "rb");
766  if (!f) return olc::NO_FILE;
767 
768  png = png_create_read_struct(PNG_LIBPNG_VER_STRING, NULL, NULL, NULL);
769  if (!png) goto fail_load;
770 
771  info = png_create_info_struct(png);
772  if (!info) goto fail_load;
773 
774  if (setjmp(png_jmpbuf(png))) goto fail_load;
775 
776  png_init_io(png, f);
777  png_read_info(png, info);
778 
779  png_byte color_type;
780  png_byte bit_depth;
781  png_bytep *row_pointers;
782  width = png_get_image_width(png, info);
783  height = png_get_image_height(png, info);
784  color_type = png_get_color_type(png, info);
785  bit_depth = png_get_bit_depth(png, info);
786 
787 #ifdef _DEBUG
788  std::cout << "Loading PNG: " << sImageFile << "\n";
789  std::cout << "W:" << width << " H:" << height << " D:" << (int)bit_depth << "\n";
790 #endif
791 
792  if (bit_depth == 16) png_set_strip_16(png);
793  if (color_type == PNG_COLOR_TYPE_PALETTE) png_set_palette_to_rgb(png);
794  if (color_type == PNG_COLOR_TYPE_GRAY && bit_depth < 8) png_set_expand_gray_1_2_4_to_8(png);
795  if (png_get_valid(png, info, PNG_INFO_tRNS)) png_set_tRNS_to_alpha(png);
796  if (color_type == PNG_COLOR_TYPE_RGB ||
797  color_type == PNG_COLOR_TYPE_GRAY ||
798  color_type == PNG_COLOR_TYPE_PALETTE)
799  png_set_filler(png, 0xFF, PNG_FILLER_AFTER);
800  if (color_type == PNG_COLOR_TYPE_GRAY ||
801  color_type == PNG_COLOR_TYPE_GRAY_ALPHA)
802  png_set_gray_to_rgb(png);
803 
804  png_read_update_info(png, info);
805  row_pointers = (png_bytep*)malloc(sizeof(png_bytep) * height);
806  for (int y = 0; y < height; y++) {
807  row_pointers[y] = (png_byte*)malloc(png_get_rowbytes(png, info));
808  }
809  png_read_image(png, row_pointers);
811 
812  // Create sprite array
813  pColData = new Pixel[width * height];
814 
815  // Iterate through image rows, converting into sprite format
816  for (int y = 0; y < height; y++)
817  {
818  png_bytep row = row_pointers[y];
819  for (int x = 0; x < width; x++)
820  {
821  png_bytep px = &(row[x * 4]);
822  SetPixel(x, y, Pixel(px[0], px[1], px[2], px[3]));
823  }
824  }
825 
826  fclose(f);
827  return olc::OK;
828 
829  fail_load:
830  width = 0;
831  height = 0;
832  fclose(f);
833  pColData = nullptr;
834  return olc::FAIL;
835 #endif

883  {
884  u = u * width - 0.5f;
885  v = v * height - 0.5f;
886  int x = (int)floor(u); // cast to int rounds toward zero, not downward

890  float u_opposite = 1 - u_ratio;
891  float v_opposite = 1 - v_ratio;
892

896  olc::Pixel p4 = GetPixel(std::min(x + 1, (int)width - 1), std::min(y + 1, (int)height - 1));
897 
898  return olc::Pixel(
899  (uint8_t)((p1.r * u_opposite + p2.r * u_ratio) * v_opposite + (p3.r * u_opposite + p4.r * u_ratio) * v_ratio),
900  (uint8_t)((p1.g * u_opposite + p2.g * u_ratio) * v_opposite + (p3.g * u_opposite + p4.g * u_ratio) * v_ratio),
901  (uint8_t)((p1.b * u_opposite + p2.b * u_ratio) * v_opposite + (p3.b * u_opposite + p4.b * u_ratio) * v_ratio));
